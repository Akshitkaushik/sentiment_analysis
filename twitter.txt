On Error Resume Next

Dim xlApp, wb, ws, lastRow, cell
Set xlApp = GetObject(, "Excel.Application") ' Attach to already open Excel

If Err.Number <> 0 Then
    MsgBox "Excel is not open.", vbCritical
    WScript.Quit
End If

Set wb = xlApp.ActiveWorkbook
Set ws = xlApp.ActiveSheet

' Find the last used row in column L
lastRow = ws.Cells(ws.Rows.Count, "L").End(-4162).Row ' -4162 = xlUp

' Loop through L2 to lastRow
For Each cell In ws.Range("L2:L" & lastRow)
    If IsDate(cell.Value) Then
        If TimeValue(cell.Value) > TimeValue("16:00") Then
            cell.Interior.Color = RGB(255, 255, 0) ' Yellow
        End If
    End If
Next

MsgBox "Highlighting complete.", vbInformation

' Cleanup
Set cell = Nothing
Set ws = Nothing
Set wb = Nothing
Set xlApp = Nothing
