Sub HighlightPostTimeGreaterThan4PM()
    On Error GoTo ErrorHandler

    Dim ws As Worksheet
    Dim lastRow As Long
    Dim cell As Range
    Dim filePath As String
    Dim wb As Workbook

    ' Set your file path here
    filePath = "C:\Users\YourName\Desktop\yourfile.xlsx"

    ' Open the workbook
    Set wb = Workbooks.Open(filePath)
    Set ws = wb.Sheets(1)

    ' Find the last used row in column L
    lastRow = ws.Cells(ws.Rows.Count, "L").End(xlUp).Row

    ' Loop through each cell in column L (starting from row 2)
    For Each cell In ws.Range("L2:L" & lastRow)
        If IsNumeric(cell.Value) Then
            ' 0.6666667 = 4:00 PM in Excel serial time
            If cell.Value > 0.6666667 Then
                cell.Interior.Color = RGB(255, 255, 0) ' Yellow
            End If
        End If
    Next cell

    MsgBox "Highlighting complete.", vbInformation
    Exit Sub

ErrorHandler:
    MsgBox "Error occurred: " & Err.Description, vbCritical
End Sub
