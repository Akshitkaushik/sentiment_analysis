Sub OpenAndHighlightExcel()
    On Error GoTo ErrorHandler

    Dim wb As Workbook
    Dim ws As Worksheet
    Dim cell As Range
    Dim lastRow As Long
    Dim filePath As String

    ' 🔁 Set your Excel file path here:
    filePath = "C:\Users\YourName\Desktop\yourfile.xlsx"

    ' 📂 Check if file exists
    If Dir(filePath) = "" Then
        MsgBox "File not found: " & filePath, vbCritical
        Exit Sub
    End If

    ' 📖 Open the workbook
    Set wb = Workbooks.Open(filePath)
    Set ws = wb.Sheets(1)

    ' 🔎 Find the last used row in column L
    lastRow = ws.Cells(ws.Rows.Count, "L").End(xlUp).Row

    ' 🔁 Loop from row 2 to last row
    For Each cell In ws.Range("L2:L" & lastRow)
        If IsDate(cell.Value) Then
            If TimeValue(cell.Value) > TimeValue("16:00") Then
                cell.EntireRow.Interior.Color = RGB(255, 255, 0) ' Highlight full row in Yellow
            End If
        End If
    Next cell

    MsgBox "Highlighting complete for Post Time > 16:00.", vbInformation
    Exit Sub

ErrorHandler:
    MsgBox "Error occurred: " & Err.Description, vbCritical
End Sub
